def ExempleA(n):
    A=np.zeros((n,n))
    for i in range(1,n-1):
        A[i,i-1], A[i,i], A[i,i+1] = -1,2,-1
    i=0
    A[i,i], A[i,i+1] = 2,-1
    i=n-1
    A[i,i-1], A[i,i] = -1,2
    return A 

n=6
X=ExempleA(n)
print(X)
\end{lstlisting}

    x, y = p[0], p[1]
    q=[exp(x)+x*y*y-1,sin(x+y)+x*x-1]
    return q

x=  fsolve(f, (0,0))
